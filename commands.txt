========================
НАВІГАЦІЯ ПО КОНСОЛІ
========================

cd "шлях до файлу"						:: зайти у папку
ls										:: список компонентів
clear									:: очистити консоль
reset									:: очистити консоль
history 								:: історія введених команд
exit									:: вийти з консолі

========================
НАЛАШТУВАННЯ GIT
========================

git --version							:: версія Git

git config --global user.name "ім`я"	:: встановити глобальне ім'я автора комітів
git config --global user.email "пошта"	:: встановити глобальну пошту автора комітів

git config --local user.name "ім`я"		:: встановити локальне ім'я автора комітів
git config --local user.email "пошта"	:: встановити локальну пошту автора комітів

git config --global --list              :: показати глобальні налаштування
git config --local --list               :: показати локальні налаштування

========================
РОБОТА З РЕПОЗИТОРІЄМ
========================

git init								:: створити репозиторій
git status								:: показати статус репозиторію

git add назва_файлу						:: додати файл в індекс
git add .								:: додати все в індекс

git commit -m "коментар"				:: створити комміт, "Init commit" - ім'я початкового комітту
git commit --amend -m "коментар"        :: змінити останній коміт

git log									:: історія комітів
		--oneline						:: історія комітів, коротко

========================
ПЕРЕГЛЯД ЗМІН
========================

git diff								:: показати зміни у файлах
		--cached						:: показати зміни в індексі

========================
СКАСУВАННЯ ЗМІН
========================

git restore назва_файлу					:: повернути файл до стану останнього коміту

git checkout -- назва_файлу				:: скасувати локальні зміни у файлі (застарілий спосіб, зараз рекомендують git restore), використовувати лише для роботи з гілками

git reset назва_файлу					:: прибрати файл з індексу
git reset --hard HEAD~2					:: видалити останні 2 коміти та їх зміни

git config --local --list				:: налаштування локального репозиторію
git config --global --list				:: налаштування глобального репозиторію

git revert HEAD                         :: створити новий коміт, який скасовує останній коміт

git rm --cached назва_файлу				:: прибрати з трекінгу

========================
ГІЛКИ (BRANCHES)
========================

git branch                    			:: показати локальні гілки
		-a								:: показати всі гілки
		ім`я_гілки                   	:: створити гілку
		-d								:: видалити гілку
		-m								:: перемістити/перейменувати гілку
		
git switch ім`я_гілки					:: перейти в гілку
		-c ім`я_гілки					:: створити гілку і перейти на неї

git checkout -b ім`я_гілки              :: старий спосіб створити гілку і перейти на неї

git merge ім`я_гілку					:: злити гілку в поточну

git rebase								:: перенести зміни поверх іншої гілки

========================
STASH ("КИШЕНЯ")
========================

git stash								:: сховати поточні зміни
		list							:: список збережених змін
		pop								:: повернути останній запис і видалити його
		apply stash@{індекс}			:: повернути конкретний запис
		clear							:: очистити stash

========================
РОБОТА З GITHUB
========================

git pull                      			:: отримати зміни з сервера

git push                      			:: відправити коміти на сервер
